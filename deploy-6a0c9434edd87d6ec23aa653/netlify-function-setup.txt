const fetch = require('node-fetch');

exports.handler = async (event) => {
    // CORS headers
    const headers = {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Headers': 'Content-Type',
        'Content-Type': 'application/json',
    };

    // Handle CORS preflight
    if (event.httpMethod === 'OPTIONS') {
        return { statusCode: 200, headers };
    }

    // Only accept POST requests
    if (event.httpMethod !== 'POST') {
        return {
            statusCode: 405,
            headers,
            body: JSON.stringify({ error: 'Method Not Allowed' }),
        };
    }

    try {
        // Parse incoming request from frontend
        const payload = JSON.parse(event.body);

        // Call Groq API with key from environment variable
        // Key is NEVER exposed to frontend
        const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${process.env.GROQ_API_KEY}`,
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(payload),
        });

        if (!response.ok) {
            const error = await response.json();
            return {
                statusCode: response.status,
                headers,
                body: JSON.stringify({
                    error: error.error?.message || 'Groq API Error',
                }),
            };
        }

        const data = await response.json();

        return {
            statusCode: 200,
            headers,
            body: JSON.stringify(data),
        };
    } catch (error) {
        console.error('Chat function error:', error);
        return {
            statusCode: 500,
            headers,
            body: JSON.stringify({
                error: 'Internal Server Error',
                message: error.message,
            }),
        };
    }
};

---END OF CODE---

HOW THE FLOW WORKS
═══════════════════════════════════════════════════════════════════════════════

USER → FRONTEND → NETLIFY FUNCTION → GROQ API → RESPONSE
                (no API key here!)   (key stored here)

1. User types message in chatbot
2. Frontend sends to: /.netlify/functions/chat
3. Netlify function receives request (no auth needed)
4. Function adds GROQ_API_KEY from environment
5. Function calls Groq API securely
6. Groq returns response
7. Function passes response back to frontend
8. Frontend displays in chat

TROUBLESHOOTING
═══════════════════════════════════════════════════════════════════════════════

ISSUE: "Failed to fetch"
CAUSE: Function not deployed
FIX: Deploy to Netlify, wait 1-2 minutes

ISSUE: "401 Unauthorized"
CAUSE: GROQ_API_KEY invalid or not set
FIX: 
1. Check Netlify environment variables
2. Verify key copied correctly from console.groq.com
3. Redeploy after adding env var

ISSUE: "404 Not Found"
CAUSE: Function path wrong
FIX: Verify path is exactly: /.netlify/functions/chat

ISSUE: "CORS error"
CAUSE: Missing CORS headers in function
FIX: Check function includes:
     'Access-Control-Allow-Origin': '*'

ISSUE: "No response received"
CAUSE: Groq didn't return expected format
FIX: Check browser DevTools Network tab
     Look at response from /.netlify/functions/chat

ISSUE: Chatbot still not working
CAUSE: ndilo-hardened.html not updated
FIX: Make sure you uploaded the NEW file
     Old file calls Groq directly (won't work)
     New file calls /.netlify/functions/chat
